Importing necessary Modules

In[]:
import numpy as np
import matplotlib.pyplot as plt

Defining the frequency range

In[]:
omega = np.linspace(-np.pi, np.pi, 1000)

Define the values of alpha

In[]:
alpha_values = [9/10, 1/2, 1/4]

Computing the frequency response for each alpha

In[]:
H = lambda alpha, omega: 1 / (1 - alpha * np.exp(-1j * omega))

Plotting the magnitude response

In[]:
plt.figure(figsize=(12, 8))

for alpha in alpha_values:
    H_alpha = H(alpha, omega)
    plt.plot(omega, np.abs(H_alpha), label=f' = {alpha}')

plt.title('Magnitude Response of the System for Different  Values')
plt.xlabel('Frequency (rad/sample)')
plt.ylabel('Magnitude')
plt.legend()
plt.grid(True)
plt.show()
